function confirmarPedido() {
  const btnCerrarPedido = document.getElementById("btn-confirmar-pedido");

  btnCerrarPedido.addEventListener("click", function() {
    const nombreValido = validateAndSetPlaceholder("nombre");
    const telValido = validateAndSetPlaceholder("tel");
    const direccionValida = validateAndSetPlaceholder("direccion");
    const callesValidas = validateAndSetPlaceholder("calles");

    if (nombreValido && telValido && direccionValida && callesValidas) {
      enviarPedido();
    }
  });

  function validateAndSetPlaceholder(elementId) {
    const element = document.getElementById(elementId);
    let isValid = true;

    if (element.value === "") {
      element.placeholder = "¡Complete el campo!";
      isValid = false;
    }

    return isValid;
  }

  
}

  alert("¡Pedido confirmado!");

}
